# ...

    # ...

views.py file

# ...
from django.shortcuts import render

# ...
# Create your views here.
# ...

def display(request):
    return render(request, 'display.html')

# ...

Important note: all this files are inside ex01 app, so we have to setup in the settings.py file located at the root of the project.

settings.py file (located at the root of the project)

#...
INSTALLED_APPS = [
    #...
    'ex01.apps.Ex01Config',
    #...
]
#...
